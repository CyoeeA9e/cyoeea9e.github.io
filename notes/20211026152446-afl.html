1 fork server

https://lcamtuf.blogspot.com/2014/10/fuzzing-binaries-without-execve.html

通常的做法是使用execve()执行程序，然后等待子进程pid。 但子进程初始化过程，包括动态库的链接、库的初始化等过程需要耗费不少的时间，并且这些过程在每次调用execve()都需要重新进行一次。

afl为了解决这个问题，在程序加载过程中进行插装，在完成了初始化过程之后，在实际执行之前，暂停程序的执行，在收到继续的信号后，调用fork来创建新的子进程。

因为有copy-on-write，fork的速度非常快，并且能够确保进程隔离。

方法有：

  • LD_PRELOAD
  • PTRACE_POKETEXT
  • 编译时插装
  • 修改ELF文件内容

问题：

  • 文件描述符

  • fuzzer变成了祖父进程，不能直接使用waitpid()和获取程序状态，因此需要通过父进程进行监控。 简单的例子 

    __afl_forkserver:

  /* Phone home and tell the parent that we're OK. */

  pushl $4          /* length    */
  pushl $__afl_temp /* data      */
  pushl $199        /* file desc */
  call  write
  addl  $12, %esp

__afl_fork_wait_loop:

  /* Wait for parent by reading from the pipe. This will block until
     the parent sends us something. Abort if read fails. */

  pushl $4          /* length    */
  pushl $__afl_temp /* data      */
  pushl $198        /* file desc */
  call  read
  addl  $12, %esp

  cmpl  $4, %eax
  jne   __afl_die

  /* Once woken up, create a clone of our process. */

  call fork

  cmpl $0, %eax
  jl   __afl_die
  je   __afl_fork_resume

  /* In parent process: write PID to pipe, then wait for child. 
     Parent will handle timeouts and SIGKILL the child as needed. */

  movl  %eax, __afl_fork_pid

  pushl $4              /* length    */
  pushl $__afl_fork_pid /* data      */
  pushl $199            /* file desc */
  call  write
  addl  $12, %esp

  pushl $2             /* WUNTRACED */
  pushl $__afl_temp    /* status    */
  pushl __afl_fork_pid /* PID       */
  call  waitpid
  addl  $12, %esp

  cmpl  $0, %eax
  jle   __afl_die

  /* Relay wait status to pipe, then loop back. */

  pushl $4          /* length    */
  pushl $__afl_temp /* data      */
  pushl $199        /* file desc */
  call  write
  addl  $12, %esp

  jmp __afl_fork_wait_loop

__afl_fork_resume:

  /* In child process: close fds, resume execution. */

  pushl $198
  call  close

  pushl $199
  call  close

  addl  $8, %esp
  ret
